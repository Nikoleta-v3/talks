\begin{frame}
    \begin{Lemma}
        An agreeable two-bit reactive strategy \(\hat{p} =
        (\hat{p}_1,\hat{p}_2,\hat{p}_3,\hat{p}_4)\) is of Nash type if and only
        if neither AllD nor the Alternator strategy yield a larger payoff.
    \end{Lemma}
\end{frame}
